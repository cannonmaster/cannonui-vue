The switch component is a reusable UI element that can be used to represent an on/off or true/false state in a web application. The code provided is a Vue 3 implementation of the switch component that can be easily integrated into a Vue application.

### Example Usages

<SwitchVue client:visible />

Here’s an example of how to use the Card component:

```vue
<script lang="ts">
import { ref } from 'vue';
export default {
  props: {
    lang: {
      type: String,
      default: 'en'
    }
  },
  emits: [],
  components: {},
  setup(props) {
    const isAsync = ref(false);
    const change = (value: Boolean, e: any) => {
      console.log('change function called');
    };
    const changeAsync = (value: Boolean, e: any) => {
      setTimeout(() => {
        isAsync.value = !isAsync.value;
      }, 2000);
    };
    const checked = ref(false);
    return {
      checked,
      isAsync,
      changeAsync,
      change
    };
  }
};
</script>

<template>
  <h2>Basic</h2>
  <cannonui-switch v-model="checked" />
  <h2>Disabled</h2>
  <cannonui-switch
    disable
    v-model="checked"
  />
  <h2>onChange</h2>
  <cannonui-switch
    v-model="checked"
    @change="change"
  />
  <h2>Async</h2>
  <cannonui-switch
    :model-Value="isAsync"
    @change="changeAsync"
  />
  <h2>Customize color</h2>
  <cannonui-switch
    v-model="checked"
    activeColor="cyan"
  />
  <h2>Customize text</h2>
  <cannonui-switch
    v-model="checked"
    activeText="开"
    inactiveText="关"
  />
</template>

<style scoped lang="scss"></style>
```

### Props

| Prop Name     | Type                       | Default Value | Description                                            |
| ------------- | -------------------------- | ------------- | ------------------------------------------------------ |
| modelValue    | String or Boolean          | false         | The current value of the switch                        |
| activeValue   | String, Number, or Boolean | true          | The value of the switch when it is active              |
| inactiveValue | String, Number, or Boolean | false         | The value of the switch when it is inactive            |
| loading       | Boolean                    | false         | Whether the switch is in a loading state               |
| disable       | Boolean                    | false         | Whether the switch is disabled                         |
| activeColor   | String                     | ""            | The background color of the switch when it is active   |
| inactiveColor | String                     | ""            | The background color of the switch when it is inactive |
| activeText    | String                     | ""            | The text to display when the switch is active          |
| inactiveText  | String                     | ""            | The text to display when the switch is inactive        |

### Events

| Event Name          | Payload                                      | Description                                                               |
| ------------------- | -------------------------------------------- | ------------------------------------------------------------------------- |
| `update:modelValue` | `String`, `Number`, or `Boolean`             | This event is emitted whenever the value of the switch component changes. |
| `change`            | `String`, `Number`, or `Boolean` and `Event` | This event is emitted whenever the value of the switch component changes. |
