Infinite loading

On this page

  • Example Usages
  • Props
  • Events

The InfiniteLoading component is a reusable Vue component that enables infinite scrolling and loading of content. The component is designed to be customizable and easy to use.

Example Usages

Infinite loading

Here’s an example of how to use the infinite component:

<style lang="scss" scoped>
.styleul {
  height: 100%;
  width: 100%;
  padding: 0;
  overflow-y: scroll;
  overflow-x: hidden;
}
.styleli {
  margin-top: 10px;
  font-size: 16px;
  height: 50px;
  text-align: center;
}
</style>
<template>
  <div class="phone">
    <div class="phone-top">Infinite loading</div>
    <div class="phone-content">
      <ul class="styleul">
        <cannonui-infiniteloading
          :loading-text="loading"
          :has-no-more-data="hasNoMore"
          :hasMore="hasMore"
          @refresh-on-pull="(completed) => refreshOnPull(completed)"
          @load-more="(completed) => loadMore(completed)"
        >
          <li
            v-for="(item, index) in defaultlist"
            :key="index"
            class="styleli"
          >
            {{ item }}
          </li>
        </cannonui-infiniteloading>
      </ul>
    </div>
    <div class="phone-bottom"></div>
  </div>
</template>
<script lang="ts">
import { onMounted, reactive, ref, toRefs } from 'vue';
export default {
  props: {
    lang: {
      type: String,
      default: 'en'
    }
  },
  emits: [],
  components: {},
  setup(props) {
    const { lang } = props;
    const data = reactive({
      defaultlist: [],
      loading: 'loading...',
      hasNoMore: 'no data'
    });
    const hasMore = ref(true);
    const loadMore = (completed: () => void) => {
      console.log(123);
      setTimeout(() => {
        const len = data.defaultlist.length;
        if (len > 30) {
          hasMore.value = false;
          completed();
          return;
        }
        for (let i = len; i < len + 10; i++) {
          data.defaultlist.push(`${i}`);
        }
        completed();
      }, 600);
    };
    const refreshOnPull = (completed: () => void) => {
      return new Promise((res, rej) => {
        setTimeout(() => {
          completed();
          res('1');
        }, 2000);
      });
    };
    onMounted(() => {
      init();
    });
    const init = () => {
      for (let i = 0; i < 10; i++) {
        data.defaultlist.push(`${i}`);
      }
    };
    return {
      refreshOnPull,
      ...toRefs(data),
      hasMore,
      loadMore
    };
  }
};
</script>

Props

Prop NameTypeDefault ValueDescription
loadingTextString‘loading…’The text to display when the component is loading more data.
hasNoMoreDataString‘no data~’The text to display when all available data has been loaded.
isRefreshOnPullBooleantrueA boolean value that determines whether the component should refresh when pulled down to the top of the page.
thresholdNumber0The distance from the bottom of the page at which the component should start loading more data.
hasMoreBooleantrueA boolean value that determines whether there is more data available to load.

Events

Event NameDescription
loadMoreTriggered when the component needs to load more data. The event should be handled by the parent component to fetch more data.
